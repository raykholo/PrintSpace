Keyboard commands

Transformation:
[s] = scale
[r] = rotate
[t] = translate
[g] = snap to grid / free move
[x] = apply transformations

Selection:
[a] = select/unselect all

Helpers:
[f] = drop
[p] = pack
[d] = remove
[j] = group
[e] = ungroup

[w] = show/hide wireframe
[h] = show / hide transformation helper

Boolean operations:
[U] = union
[S] = subtract
[I] = intersect

Export STL:
[n] = save binary
[m] = save ascii

Views:
[0] = default
[2] = front
[6] = right
[8] = back
[4] = left
[9] = top
[3] = bottom
